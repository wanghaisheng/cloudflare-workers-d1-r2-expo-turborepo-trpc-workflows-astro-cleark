# 通用虚假实现防范

## 🎯 目标

通过.codex配置强化虚假实现防范，确保：
1. 任务定义必须明确且可验证
2. 数据源必须真实且可访问
3. 实现必须完整且可测试
4. 验收必须严格且可重复

## 🎯 通用检测脚本

### 1. **任务定义检测**
```python
# .codex/scripts/task_definition_detector.py
import yaml
import re
from pathlib import Path

class TaskDefinitionDetector:
    """通用任务定义检测器"""
    
    def __init__(self):
        self.validation_rules = self._load_validation_rules()
    
    def detect_issues(self, task_file):
        """检测任务定义问题"""
        issues = []
        
        # 1. 检查必需字段
        missing_fields = self._check_required_fields(task_file)
        if missing_fields:
            issues.append(f"缺少必需字段: {missing_fields}")
        
        # 2. 检查任务ID格式
        if not self._check_task_id_format(task_file):
            issues.append("任务ID格式不正确")
        
        # 3. 检查验收标准
        if not self._check_acceptance_criteria(task_file):
            issues.append("验收标准不明确或不可验证")
        
        # 4. 检查实现指南
        if not self._check_implementation_guide(task_file):
            issues.append("实现指南不明确或不可操作")
        
        return issues
    
    def _check_required_fields(self, task_file):
        """检查必需字段"""
        content = task_file.read_text()
        missing_fields = []
        
        required_fields = self.validation_rules['task_definition_validation']['required_fields']
        for field in required_fields:
            if field not in content:
                missing_fields.append(field)
        
        return missing_fields
    
    def _check_task_id_format(self, task_file):
        """检查任务ID格式"""
        content = task_file.read_text()
        task_id_pattern = r'Task \d+\.\d+:'
        return re.search(task_id_pattern, content) is not None
    
    def _check_acceptance_criteria(self, task_file):
        """检查验收标准"""
        content = task_file.read_text()
        
        # 检查是否有明确的验收标准
        criteria_pattern = r'- \[.\] .*: (✅|❌|⚠️) .*'
        criteria_matches = re.findall(criteria_pattern, content)
        
        return len(criteria_matches) >= 3
    
    def _check_implementation_guide(self, task_file):
        """检查实现指南"""
        content = task_file.read_text()
        
        # 检查是否有实现步骤
        steps_pattern = r'\d+\..+:'
        steps_matches = re.findall(steps_pattern, content)
        
        return len(steps_matches) >= 3
```

### 2. **虚假实现检测**
```python
# .codex/scripts/fake_implementation_detector.py
import ast
import re
from pathlib import Path

class FakeImplementationDetector:
    """通用虚假实现检测器"""
    
    def __init__(self):
        self.fake_patterns = self._load_fake_patterns()
    
    def detect_fake_implementation(self, implementation_dir):
        """检测虚假实现"""
        issues = []
        
        # 1. 检查模拟数据使用
        mock_data_issues = self._check_mock_data_usage(implementation_dir)
        issues.extend(mock_data_issues)
        
        # 2. 检查不完整实现
        incomplete_implementation_issues = self._check_incomplete_implementation(implementation_dir)
        issues.extend(incomplete_implementation_issues)
        
        # 3. 检查缺失核心功能
        missing_core_function_issues = self._check_missing_core_functions(implementation_dir)
        issues.extend(missing_core_function_issues)
        
        # 4. 检查虚假验收
        fake_validation_issues = self._check_fake_validation(implementation_dir)
        issues.extend(fake_validation_issues)
        
        return issues
    
    def _check_mock_data_usage(self, implementation_dir):
        """检查模拟数据使用"""
        issues = []
        
        for py_file in implementation_dir.rglob("*.py"):
            content = py_file.read_text()
            
            for pattern in self.fake_patterns['mock_data_patterns']:
                if pattern in content:
                    issues.append(f"{py_file}: 发现模拟数据使用 - {pattern}")
        
        return issues
    
    def _check_incomplete_implementation(self, implementation_dir):
        """检查不完整实现"""
        issues = []
        
        for py_file in implementation_dir.rglob("*.py"):
            content = py_file.read_text()
            
            for pattern in self.fake_patterns['incomplete_patterns']:
                if pattern in content:
                    issues.append(f"{py_file}: 发现不完整实现 - {pattern}")
        
        return issues
    
    def _check_missing_core_functions(self, implementation_dir):
        """检查缺失核心功能"""
        issues = []
        
        for py_file in implementation_dir.rglob("*.py"):
            try:
                tree = ast.parse(py_file.read_text())
                defined_functions = [node.name for node in ast.walk(tree) if isinstance(node, ast.FunctionDef)]
                
                # 检查通用核心函数
                core_functions = ['initialize', 'process', 'validate', 'cleanup']
                for core_func in core_functions:
                    if core_func not in defined_functions:
                        issues.append(f"{py_file}: 缺失核心函数 - {core_func}")
            except SyntaxError:
                issues.append(f"{py_file}: 语法错误，无法解析")
        
        return issues
    
    def _check_fake_validation(self, implementation_dir):
        """检查虚假验收"""
        issues = []
        
        for py_file in implementation_dir.rglob("*.py"):
            content = py_file.read_text()
            
            for pattern in self.fake_patterns['fake_validation_patterns']:
                if pattern in content:
                    issues.append(f"{py_file}: 发现虚假验收 - {pattern}")
        
        return issues
```

### 3. **自动化验证脚本**
```python
# .codex/scripts/automated_validator.py
import subprocess
import json
from pathlib import Path

class AutomatedValidator:
    """自动化验证器"""
    
    def __init__(self):
        self.task_detector = TaskDefinitionDetector()
        self.fake_detector = FakeImplementationDetector()
    
    def validate_task_implementation(self, task_file, implementation_dir):
        """验证任务实现"""
        results = {
            'task_definition': {},
            'implementation': {},
            'validation': {},
            'overall_status': 'pending'
        }
        
        # 1. 验证任务定义
        task_issues = self.task_detector.detect_issues(task_file)
        results['task_definition'] = {
            'issues': task_issues,
            'status': 'passed' if len(task_issues) == 0 else 'failed'
        }
        
        # 2. 验证实现
        impl_issues = self.fake_detector.detect_fake_implementation(implementation_dir)
        results['implementation'] = {
            'issues': impl_issues,
            'status': 'passed' if len(impl_issues) == 0 else 'failed'
        }
        
        # 3. 运行验证测试
        validation_results = self._run_validation_tests(implementation_dir)
        results['validation'] = validation_results
        
        # 4. 计算总体状态
        all_passed = all([
            results['task_definition']['status'] == 'passed',
            results['implementation']['status'] == 'passed',
            results['validation']['status'] == 'passed'
        ])
        
        results['overall_status'] = 'passed' if all_passed else 'failed'
        
        return results
    
    def _run_validation_tests(self, implementation_dir):
        """运行验证测试"""
        try:
            # 运行单元测试
            unit_test_result = subprocess.run(
                ['python', '-m', 'pytest', 'tests/unit/', '-v'],
                capture_output=True,
                text=True,
                cwd=implementation_dir
            )
            
            # 运行集成测试
            integration_test_result = subprocess.run(
                ['python', '-m', 'pytest', 'tests/integration/', '-v'],
                capture_output=True,
                text=True,
                cwd=implementation_dir
            )
            
            return {
                'unit_tests': {
                    'status': 'passed' if unit_test_result.returncode == 0 else 'failed',
                    'output': unit_test_result.stdout,
                    'errors': unit_test_result.stderr
                },
                'integration_tests': {
                    'status': 'passed' if integration_test_result.returncode == 0 else 'failed',
                    'output': integration_test_result.stdout,
                    'errors': integration_test_result.stderr
                },
                'status': 'passed' if (
                    unit_test_result.returncode == 0 and 
                    integration_test_result.returncode == 0
                ) else 'failed'
            }
        except Exception as e:
            return {
                'status': 'failed',
                'error': str(e)
            }
```
